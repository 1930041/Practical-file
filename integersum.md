//Program to display the Sum of first 100 positive integers

#include<stdio.h>  
int main()              {  
  
int sum=0,n;  
n=1;  
while(n<=100)  
{  
sum+=n;  
n++;  
}  
printf("Sum of first 100 +ve integers = %d\n",sum);  
  
return 0;  
}
